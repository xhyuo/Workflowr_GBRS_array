lm_eqn <- function(df){
    df$x <- as.numeric(as.character(df$x))
    m <- lm(y ~ x, df);
    eq <- substitute(italic(y) == a + b %.% italic(x), 
         list(a = format(coef(m)[1], digits = 2), 
              b = format(coef(m)[2], digits = 2)))
    as.character(as.expression(eq));                 
  }
